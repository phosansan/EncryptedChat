80       private void kirimPesan(String text) {
81           Date date = Calendar.getInstance().getTime();
82           SimpleDateFormat tanggal = new SimpleDateFormat("dd-MM-yyyy");
83           String hari = tanggal.format(date);
84   
85           Calendar calendar = Calendar.getInstance();
86           SimpleDateFormat pukul = new SimpleDateFormat("hh:mm");
87           String jam = pukul.format(calendar.getTime());
88   
89           Pesan pesan = new Pesan(text,
90                   firebaseUser.getUid(),
91                   IDPenerima,
92                   hari + " , " + jam);
93   
94           reference.child("Pesan").push().setValue(pesan).addOnSuccessListener(new OnSuccessListener<Void>() {
95               @Override
96               public void onSuccess(Void aVoid) {
97                   Log.d("Pengiriman", "Berhasil");
98               }
99           });
100  
101          DatabaseReference reference1 = FirebaseDatabase.getInstance().getReference("Daftar Chat").
102                  child(firebaseUser.getUid()).child(IDPenerima);
103          reference1.child("IDCHat").setValue(IDPenerima);
104  
105          DatabaseReference reference2 = FirebaseDatabase.getInstance().getReference("Daftar Chat").
106                  child(IDPenerima).child(firebaseUser.getUid());
107          reference1.child("IDCHat").setValue(IDPenerima);
108  
109      }

111      private void bacaPesan() {
112          try {
113              DatabaseReference reference = FirebaseDatabase.getInstance().getReference();
114              reference.child("Pesan").addValueEventListener(new ValueEventListener() {
115                  @Override
116                  public void onDataChange(@NonNull DataSnapshot dataSnapshot) {
117                      list.clear();
118                      for (DataSnapshot snapshot : dataSnapshot.getChildren()){
119                          Pesan pesan = snapshot.getValue(Pesan.class);
120                          if (pesan != null
121                                  && pesan.getPengirim().equals(firebaseUser.getUid())
122                                  && pesan.getPenerima().equals(IDPenerima)
123                                  || pesan.getPenerima().equals(firebaseUser.getUid())
124                                  && pesan.getPengirim().equals(IDPenerima)){
125  
126                              list.add(pesan);
127                          }
128                      }
129                      if (adapter != null) {
130                          adapter.notifyDataSetChanged();
131                      } else {
132                          adapter = new AdapterPesan(list, Chatting.this);
133                          binding.recyclerView.setAdapter(adapter);
134                      }
135                  }
136  
137                  @Override
138                  public void onCancelled(DatabaseError error) {
139  
140                  }
141              });
142  
143          } catch (Exception e) {
144              e.printStackTrace();
145          }
146      }
